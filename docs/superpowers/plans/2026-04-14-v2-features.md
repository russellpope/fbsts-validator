package browser

import (
	"os/exec"
	"runtime"
)

// Open attempts to open the given URL in the user's default browser.
// It is non-blocking and silently ignores failures. If the platform
// is unsupported or the browser command is not found, the call is a no-op.
func Open(url string) {
	cmd, args := browserCommand(url)
	if cmd == "" {
		return
	}
	// Fire and forget — don't block the caller, don't report errors.
	// The verification URL is always printed to stdout as a fallback.
	exec.Command(cmd, args...).Start()
}

// browserCommand returns the platform-appropriate command and arguments
// for opening a URL. Returns ("", nil) on unsupported platforms.
func browserCommand(url string) (string, []string) {
	switch runtime.GOOS {
	case "darwin":
		return "open", []string{url}
	case "linux":
		return "xdg-open", []string{url}
	case "windows":
		return "rundll32", []string{"url.dll,FileProtocolHandler", url}
	default:
		return "", nil
	}
}
```

package runner

import (
	"fmt"
	"time"

	"github.com/pure-experimental/rp-fbstsvalidator/internal/render"
	"github.com/pure-experimental/rp-fbstsvalidator/internal/steps"
)

// Runner orchestrates the step pipeline.
type Runner struct {
	renderer render.Renderer
	// DemoPace adds a delay between step completions for presentation effect.
	// Zero means no delay.
	DemoPace time.Duration
}

// New creates a Runner with the given renderer and no demo pacing.
func New(renderer render.Renderer) *Runner {
	return &Runner{renderer: renderer}
}

// Run executes each step in order. If continueOnError is false, it stops at
// the first failure. Returns an error if any step failed.
func (r *Runner) Run(ctx *steps.FlowContext, pipeline []steps.Step, continueOnError bool) error {
	results := make(map[string]*steps.StepResult)
	var order []string
	var firstErr error

	for i, step := range pipeline {
		name := step.Name()
		order = append(order, name)

		// Demo pacing: pause before starting the next step (not before the first).
		if r.DemoPace > 0 && i > 0 {
			time.Sleep(r.DemoPace)
		}

		r.renderer.RenderStepStart(name)

		start := time.Now()
		result, err := step.Execute(ctx)
		elapsed := time.Since(start)

		if err != nil {
			r.renderer.RenderStepError(name, err)

			if firstErr == nil {
				firstErr = fmt.Errorf("step %s failed: %w", name, err)
			}

			if !continueOnError {
				return firstErr
			}
			continue
		}

		if result.Duration == 0 {
			result.Duration = elapsed
		}
		results[name] = result
		r.renderer.RenderStepResult(name, result)
	}

	r.renderer.RenderSummary(results, order)

	return firstErr
}
```

package render

import (
	"errors"
	"fmt"
	"strings"
	"time"

	"github.com/charmbracelet/lipgloss"
	"github.com/pure-experimental/rp-fbstsvalidator/internal/steps"
)

// Station status constants for the subway model.
type stationStatus int

const (
	stationPending stationStatus = iota
	stationRunning
	stationComplete
	stationFailed
)

// station represents one node on the subway map.
type station struct {
	name   string
	status stationStatus
	result *steps.StepResult
	err    error
}

// subwayModel holds the state for the subway-map view.
// It is used both by the bubbletea program and directly for testing.
type subwayModel struct {
	stations []station
	summary  *summaryData
	warning  string
}

type summaryData struct {
	results map[string]*steps.StepResult
	order   []string
}

// Styles for the subway renderer.
var (
	stationGreen  = lipgloss.NewStyle().Foreground(lipgloss.Color("10")).Bold(true)
	stationYellow = lipgloss.NewStyle().Foreground(lipgloss.Color("11")).Bold(true)
	stationGray   = lipgloss.NewStyle().Foreground(lipgloss.Color("8"))
	stationRed    = lipgloss.NewStyle().Foreground(lipgloss.Color("9")).Bold(true)
	detailLabel   = lipgloss.NewStyle().Foreground(lipgloss.Color("8")).Width(20)
	detailValue   = lipgloss.NewStyle().Foreground(lipgloss.Color("15"))
	trackGreen    = lipgloss.NewStyle().Foreground(lipgloss.Color("10"))
	trackYellow   = lipgloss.NewStyle().Foreground(lipgloss.Color("11"))
	trackGray     = lipgloss.NewStyle().Foreground(lipgloss.Color("8"))
	trackRed      = lipgloss.NewStyle().Foreground(lipgloss.Color("9"))
	summaryTitle  = lipgloss.NewStyle().Bold(true).Foreground(lipgloss.Color("12"))
)

func newSubwayModel(stepNames []string) subwayModel {
	stations := make([]station, len(stepNames))
	for i, name := range stepNames {
		stations[i] = station{name: name, status: stationPending}
	}
	return subwayModel{stations: stations}
}

func (m subwayModel) applyStart(name string) subwayModel {
	for i := range m.stations {
		if m.stations[i].name == name {
			m.stations[i].status = stationRunning
			break
		}
	}
	return m
}

func (m subwayModel) applyResult(name string, result *steps.StepResult) subwayModel {
	for i := range m.stations {
		if m.stations[i].name == name {
			m.stations[i].status = stationComplete
			m.stations[i].result = result
			break
		}
	}
	return m
}

func (m subwayModel) applyError(name string, err error) subwayModel {
	for i := range m.stations {
		if m.stations[i].name == name {
			m.stations[i].status = stationFailed
			m.stations[i].err = err
			break
		}
	}
	return m
}

func (m subwayModel) applySummary(results map[string]*steps.StepResult, order []string) subwayModel {
	m.summary = &summaryData{results: results, order: order}
	return m
}

// renderView produces the full subway-map terminal output.
func (m subwayModel) renderView() string {
	var sb strings.Builder

	if m.warning != "" {
		sb.WriteString(stationYellow.Render("⚠  "+m.warning) + "\n\n")
	}

	for i, s := range m.stations {
		// Determine styles based on status
		var icon, nameStr, trackStyle string
		switch s.status {
		case stationComplete:
			icon = stationGreen.Render("◉")
			track := stationGreen.Render("━━")
			nameStr = fmt.Sprintf("%s %s", track, stationGreen.Render(s.name))
			trackStyle = "green"
		case stationRunning:
			icon = stationYellow.Render("◎")
			track := stationYellow.Render("━━")
			nameStr = fmt.Sprintf("%s %s %s", track, stationYellow.Render(s.name), stationYellow.Render("running..."))
			trackStyle = "yellow"
		case stationFailed:
			icon = stationRed.Render("✗")
			track := stationRed.Render("━━")
			nameStr = fmt.Sprintf("%s %s %s", track, stationRed.Render(s.name), stationRed.Render("FAILED"))
			trackStyle = "red"
		default:
			icon = stationGray.Render("○")
			track := stationGray.Render("━━")
			nameStr = fmt.Sprintf("%s %s", track, stationGray.Render(s.name))
			trackStyle = "gray"
		}

		// Duration suffix for completed stations
		durStr := ""
		if s.result != nil && s.result.Duration > 0 {
			durStr = " " + stationGray.Render(fmt.Sprintf("··· %s", s.result.Duration.Round(time.Millisecond)))
		}

		sb.WriteString(fmt.Sprintf("%s%s%s\n", icon, nameStr, durStr))

		// Detail fields for completed stations
		if s.result != nil {
			trackChar := trackColorChar(trackStyle)
			for _, f := range s.result.Fields {
				if f.Label == "" {
					continue
				}
				val := MaskField(f.Value, f.Sensitive)
				sb.WriteString(fmt.Sprintf("%s    %s  %s\n", trackChar, detailLabel.Render(f.Label), detailValue.Render(val)))
			}

			// Sub-steps as branch stations
			for _, sub := range s.result.SubSteps {
				var subIcon, subName string
				if sub.Status == steps.StatusPass {
					subIcon = stationGreen.Render("◉")
					subName = stationGreen.Render(sub.Name)
				} else if sub.Status == steps.StatusFail {
					subIcon = stationRed.Render("✗")
					subName = stationRed.Render(sub.Name)
				} else {
					subIcon = stationGray.Render("○")
					subName = stationGray.Render(sub.Name)
				}
				subDur := ""
				if sub.Duration > 0 {
					subDur = stationGray.Render(fmt.Sprintf(" (%s)", sub.Duration.Round(time.Millisecond)))
				}
				errStr := ""
				if sub.Error != "" {
					errStr = " " + stationRed.Render(sub.Error)
				}
				sb.WriteString(fmt.Sprintf("%s    %s─ %s%s%s\n", trackChar, subIcon, subName, subDur, errStr))
			}
		}

		// Error details for failed stations
		if s.status == stationFailed && s.err != nil {
			trackChar := trackColorChar(trackStyle)
			var stepErr *steps.StepError
			if errors.As(s.err, &stepErr) {
				if stepErr.Code != "" {
					sb.WriteString(fmt.Sprintf("%s    %s  %s\n", trackChar, detailLabel.Render("Error"), detailValue.Render(stepErr.Code)))
				}
				sb.WriteString(fmt.Sprintf("%s    %s  %s\n", trackChar, detailLabel.Render("Message"), detailValue.Render(stepErr.Err.Error())))
				if stepErr.Hint != "" {
					sb.WriteString(fmt.Sprintf("%s    %s  %s\n", trackChar, detailLabel.Render("Hint"), styleHint.Render(stepErr.Hint)))
				}
			} else {
				sb.WriteString(fmt.Sprintf("%s    %s  %s\n", trackChar, detailLabel.Render("Error"), detailValue.Render(s.err.Error())))
			}
		}

		// Track connector to next station
		if i < len(m.stations)-1 {
			trackChar := trackColorChar(trackStyle)
			sb.WriteString(trackChar + "\n")
		}
	}

	// Summary section
	if m.summary != nil {
		sb.WriteString("\n" + summaryTitle.Render("Validation Summary") + "\n")
		var total time.Duration
		for _, key := range m.summary.order {
			r, ok := m.summary.results[key]
			if !ok {
				sb.WriteString(fmt.Sprintf("  %s %s\n", stationRed.Render("✗"), key))
				continue
			}
			sb.WriteString(fmt.Sprintf("  %s %s %s\n",
				stationGreen.Render("✓"),
				r.Title,
				stationGray.Render(fmt.Sprintf("(%s)", r.Duration.Round(time.Millisecond)))))
			total += r.Duration
		}
		sb.WriteString(fmt.Sprintf("\n  Total: %s\n", total.Round(time.Millisecond)))
	}

	return sb.String()
}

func trackColorChar(style string) string {
	switch style {
	case "green":
		return trackGreen.Render("┃")
	case "yellow":
		return trackYellow.Render("┃")
	case "red":
		return trackRed.Render("┃")
	default:
		return trackGray.Render("┃")
	}
}

// --- SubwayRenderer: bubbletea-backed Renderer implementation ---

// SubwayRenderer implements the Renderer interface using a bubbletea TUI.
// The renderer methods send messages to the bubbletea program which updates
// the subway model and re-renders the view.
type SubwayRenderer struct {
	model    subwayModel
	program  *tea.Program
	msgCh    chan tea.Msg
	doneCh   chan struct{}
	started  bool
}

// bubbletea message types
type msgStepStart struct{ name string }
type msgStepResult struct {
	name   string
	result *steps.StepResult
}
type msgStepError struct {
	name string
	err  error
}
type msgSummary struct {
	results map[string]*steps.StepResult
	order   []string
}
type msgWarning struct{ text string }
type msgQuit struct{}

// teaModel wraps subwayModel for the bubbletea interface.
type teaModel struct {
	subway subwayModel
	msgCh  chan tea.Msg
	done   bool
}

func (t teaModel) Init() tea.Cmd {
	return t.waitForMsg()
}

func (t teaModel) waitForMsg() tea.Cmd {
	return func() tea.Msg {
		msg, ok := <-t.msgCh
		if !ok {
			return msgQuit{}
		}
		return msg
	}
}

func (t teaModel) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch m := msg.(type) {
	case msgStepStart:
		t.subway = t.subway.applyStart(m.name)
		return t, t.waitForMsg()
	case msgStepResult:
		t.subway = t.subway.applyResult(m.name, m.result)
		return t, t.waitForMsg()
	case msgStepError:
		t.subway = t.subway.applyError(m.name, m.err)
		return t, t.waitForMsg()
	case msgSummary:
		t.subway = t.subway.applySummary(m.results, m.order)
		return t, t.waitForMsg()
	case msgWarning:
		t.subway.warning = m.text
		return t, t.waitForMsg()
	case msgQuit:
		t.done = true
		return t, tea.Quit
	}
	return t, nil
}

func (t teaModel) View() string {
	return t.subway.renderView()
}

// NewSubwayRenderer creates a SubwayRenderer with the given step names.
// Call Start() before using RenderStepStart/Result/Error, and Stop() when done.
func NewSubwayRenderer(stepNames []string) *SubwayRenderer {
	return &SubwayRenderer{
		model:  newSubwayModel(stepNames),
		msgCh:  make(chan tea.Msg, 16),
		doneCh: make(chan struct{}),
	}
}

// Start launches the bubbletea program in a background goroutine.
func (r *SubwayRenderer) Start() {
	tm := teaModel{subway: r.model, msgCh: r.msgCh}
	r.program = tea.NewProgram(tm, tea.WithAltScreen())
	r.started = true

	go func() {
		r.program.Run()
		close(r.doneCh)
	}()
}

// Stop signals the bubbletea program to quit and waits for it to finish.
func (r *SubwayRenderer) Stop() {
	close(r.msgCh)
	if r.started {
		<-r.doneCh
	}
}

func (r *SubwayRenderer) RenderStepStart(name string) {
	r.msgCh <- msgStepStart{name: name}
}

func (r *SubwayRenderer) RenderStepResult(name string, result *steps.StepResult) {
	r.msgCh <- msgStepResult{name: name, result: result}
}

func (r *SubwayRenderer) RenderStepError(name string, err error) {
	r.msgCh <- msgStepError{name: name, err: err}
}

func (r *SubwayRenderer) RenderSummary(results map[string]*steps.StepResult, order []string) {
	r.msgCh <- msgSummary{results: results, order: order}
	// Give the TUI a moment to render the summary before quitting
	time.Sleep(100 * time.Millisecond)
}

func (r *SubwayRenderer) RenderWarning(msg string) {
	r.msgCh <- msgWarning{text: msg}
}
